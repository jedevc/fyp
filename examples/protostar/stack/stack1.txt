extern flag: fn () void,
       tryagain: fn () void

chunk buffer: [64]char,
      modified: int = 0

block main {
    if argc <= 1 {
        puts@libc.stdio("no argument specified")
        exit@libc.stdlib(1)
    }
    strcpy@libc.string(buffer, argv[1])
    ...

    if modified == 0x61626364 {
        flag()
    } else {
        tryagain()
    }
}
