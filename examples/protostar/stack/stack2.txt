extern flag: fn () void,
       tryagain: fn () void

chunk variable: *char
chunk buffer: [64]char,
      modified: int = 0

block main {
    variable = getenv@libc.stdlib("GREENIE")
    if variable == NULL {
        puts@libc.stdio("please set GREENIE")
        exit@libc.stdlib(1)
    }
    ...

    strcpy@libc.string(buffer, variable)
    if modified == 0x0d0a0d0a {
        flag()
    } else {
        tryagain()
    }
}
