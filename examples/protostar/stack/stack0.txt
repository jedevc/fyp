extern flag: fn () void,
       tryagain: fn () void

chunk buffer: [64]char,
      modified: int = 0

block main {
    puts@libc.stdio("Can you get the flag?")
    ...

    gets@libc.stdio(buffer)
    ...

    if modified != 0 {
        flag()
    } else {
        tryagain()
    }
}
